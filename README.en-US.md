## 📦 Installation

```bash
$ npm install @antv/l7 @antv/larkmap
```

## 🔨 Usage

```jsx
import React from 'react';
import { LarkMap } from '@antv/larkmap';

const config = {
  mapType: 'Mapbox',
  mapOptions: {
    style: 'light',
    center: [120.210792, 30.246026],
    pitch: 0,
    zoom: 8,
    rotation: 0,
  },
};

export default () => {
  return <LarkMap {...config} style={{ height: '300px' }}></LarkMap>;
};
```
